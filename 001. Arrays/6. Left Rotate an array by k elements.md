
## A. Brute Force Solution

```cpp
void Rotatetoright(int arr[], int n, int k)
{
  if (n == 0) //if n is equal to zero, simply return 
    return;
  k = k % n; 
  if (k > n)
    return;
  int temp[k];
  for (int i = n - k; i < n; i++)
  {
    temp[i - n + k] = arr[i];
  }
  for (int i = n - k - 1; i >= 0; i--)
  {
    arr[i + k] = arr[i];
  }
  for (int i = 0; i < k; i++)
  {
    arr[i] = temp[i];
  }
}
```
