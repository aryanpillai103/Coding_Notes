
## Brute Force Approach

```c++
class Solution {
  public:
  //function to find largest element in the array
    int largest(vector<int> &arr) {
        int maximum = arr[0];//declaring max variable as 
        for(int i = 0 ; i < arr.size() ; i++){
            maximum = max(maximum , arr[i]);
        }
        return maximum;
    }
};
```
