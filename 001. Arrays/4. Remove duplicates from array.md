
## Brute Force Solution

```cpp
int removeDuplicates(int arr[], int n) {
  set < int > set; //declare set
  for (int i = 0; i < n; i++) { // iterate from 1 to n
    set.insert(arr[i]); // insert array elem
  }
  int k = set.size();
  int j = 0;
  for (int x: set) {
    arr[j++] = x;
  }
  return k;
}
```
