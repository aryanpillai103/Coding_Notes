
## A. Brute force 

```cpp
void solve(int arr[], int n) {
  int temp[n]; // declare a dummy array of size n
  for (int i = 1; i < n; i++) { // iterate from 1 to n
    temp[i - 1] = arr[i]; // 
  }
  temp[n - 1] = arr[0];
  for (int i = 0; i < n; i++) {
    cout << temp[i] << " ";
  }
  cout << endl;
}
```
